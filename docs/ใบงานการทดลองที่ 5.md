# การทดลองที่ 5.1 การต่อวงจรระบบ Alarm

## ขั้นตอนการทดลอง

1. ศึกษา Scenario การแจ้งเตือนของสถานีฐาน

   - **Cabinet Door Open** — ตรวจจับด้วย Obstacle Sensor (จำลองการเปิดประตูตู้)
   - **Motion Detection** — ตรวจจับด้วย PIR Sensor (บุคคลไม่ได้รับอนุญาต)
   - **Environmental Alarm** — อุณหภูมิ/ความชื้นเกินค่ากำหนด (จาก Lab 1)

2. ต่อวงจรตามตารางต่อไปนี้

   | ESP32 GPIO | อุปกรณ์ | หมายเหตุ |
   | ---------- | ------- | -------- |
   | 3.3V | VCC (PIR, Obstacle, OLED) | จ่ายไฟ |
   | GND | GND (PIR, Obstacle, OLED, Buzzer, Relay) | กราวด์ร่วม |
   | GPIO 4 | OUT (PIR) | Digital Input |
   | GPIO 5 | OUT (Obstacle) | Digital Input |
   | GPIO 21 | SDA (OLED) | I2C Data |
   | GPIO 22 | SCL (OLED) | I2C Clock |
   | GPIO 18 | Buzzer | Digital Output |
   | GPIO 19 | Relay | Digital Output |
   | GPIO 2 | LED เขียว | Normal |
   | GPIO 15 | LED เหลือง | Warning |
   | GPIO 13 | LED แดง | Alarm |

3. ตรวจสอบการต่อวงจรให้ถูกต้อง ก่อนจ่ายไฟ

# การทดลองที่ 5.2 การจำลองเหตุการณ์และตรวจจับ

## ขั้นตอนการทดลอง

1. เขียนโปรแกรมตรวจจับเหตุการณ์

   ```cpp
   #define PIR_PIN 4
   #define OBSTACLE_PIN 5
   #define BUZZER_PIN 18
   #define RELAY_PIN 19
   #define LED_GREEN 2
   #define LED_YELLOW 15
   #define LED_RED 13

   enum AlarmLevel { NORMAL, MINOR, MAJOR, CRITICAL };
   AlarmLevel currentAlarm = NORMAL;

   void setup() {
     Serial.begin(115200);
     pinMode(PIR_PIN, INPUT);
     pinMode(OBSTACLE_PIN, INPUT);
     pinMode(BUZZER_PIN, OUTPUT);
     pinMode(RELAY_PIN, OUTPUT);
     pinMode(LED_GREEN, OUTPUT);
     pinMode(LED_YELLOW, OUTPUT);
     pinMode(LED_RED, OUTPUT);
     // Initialize OLED...
   }

   AlarmLevel evaluateAlarm() {
     bool motion = digitalRead(PIR_PIN);
     bool doorOpen = digitalRead(OBSTACLE_PIN);
     // float temp = readTemperature(); // from Lab 1

     if (motion && doorOpen) return CRITICAL;
     if (motion || doorOpen) return MAJOR;
     // if (temp > 35.0) return MINOR;
     return NORMAL;
   }

   void handleAlarm(AlarmLevel level) {
     digitalWrite(LED_GREEN, LOW);
     digitalWrite(LED_YELLOW, LOW);
     digitalWrite(LED_RED, LOW);
     digitalWrite(BUZZER_PIN, LOW);
     digitalWrite(RELAY_PIN, LOW);

     switch (level) {
       case NORMAL:
         digitalWrite(LED_GREEN, HIGH);
         break;
       case MINOR:
         digitalWrite(LED_YELLOW, HIGH);
         tone(BUZZER_PIN, 1000, 500);
         break;
       case MAJOR:
         digitalWrite(LED_YELLOW, HIGH);
         digitalWrite(LED_RED, HIGH);
         tone(BUZZER_PIN, 2000, 1000);
         break;
       case CRITICAL:
         digitalWrite(LED_RED, HIGH);
         digitalWrite(RELAY_PIN, HIGH);
         tone(BUZZER_PIN, 3000, 2000);
         break;
     }
   }

   void loop() {
     currentAlarm = evaluateAlarm();
     handleAlarm(currentAlarm);
     // Update OLED with alarm info...
     delay(500);
   }
   ```

2. ทดสอบจำลองเหตุการณ์แต่ละประเภท

   | การกระทำ | PIR | Obstacle | Alarm Level ที่คาดหวัง |
   | --------- | --- | -------- | --------------------- |
   | ไม่มีอะไร | LOW | LOW | NORMAL |
   | ยื่นมือใกล้ PIR | HIGH | LOW | MAJOR |
   | บัง Obstacle Sensor | LOW | HIGH | MAJOR |
   | ยื่นมือ + บัง Obstacle | HIGH | HIGH | CRITICAL |

# การทดลองที่ 5.3 การจำแนกระดับ Alarm และแจ้งเตือนผ่าน OLED

## ขั้นตอนการทดลอง

1. แสดงผล Alarm Status บน OLED

   ```cpp
   void displayAlarm(AlarmLevel level, const char* event) {
     display.clearDisplay();
     display.setCursor(0, 0);
     display.println("SITE A-01 ALARM");
     display.println("================");

     switch (level) {
       case NORMAL:
         display.println("STATUS: NORMAL");
         break;
       case MINOR:
         display.println("STATUS: MINOR");
         display.println("Event: " + String(event));
         break;
       case MAJOR:
         display.println("STATUS: MAJOR");
         display.println("Event: " + String(event));
         break;
       case CRITICAL:
         display.println("STATUS: CRITICAL");
         display.println("Event: " + String(event));
         display.println("ACTION REQUIRED!");
         break;
     }
     display.display();
   }
   ```

2. ทดสอบให้ครบทั้ง 4 ระดับ Alarm

### คำถามท้ายการทดลอง

1. การจำแนกระดับ Alarm เป็น Minor/Major/Critical มีประโยชน์ต่อการดำเนินงานของ NOC อย่างไร
2. ในสถานการณ์จริง ถ้ามี Alarm ระดับ Critical เกิดขึ้น ควรมีขั้นตอนการตอบสนองอย่างไร
3. Relay Module ในระบบนี้ใช้ควบคุมอุปกรณ์อะไรได้บ้างในสถานีฐานจริง

# คำถามท้ายใบงาน

1. ระบบแจ้งเตือนเหตุผิดปกติ (Alarm System) มีความสำคัญต่อสถานีฐานโทรคมนาคมอย่างไร
2. จงอธิบายความแตกต่างระหว่าง Intrusion Detection และ Fault Detection
3. ถ้าต้องการส่ง Alarm ไปยัง NOC ผ่าน MQTT ควรออกแบบ Payload อย่างไร (ยกตัวอย่าง JSON)
4. Alarm Flooding คืออะไร และมีวิธีป้องกันอย่างไรในระบบจริง
