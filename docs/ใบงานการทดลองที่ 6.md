# การทดลองที่ 6.1 การรวบรวมข้อมูล Sensor เพื่อสร้าง Dataset

## ขั้นตอนการทดลอง

1. สร้าง Data Logger บน Raspberry Pi เพื่อบันทึกข้อมูล Sensor จาก ESP32

   ```python
   import paho.mqtt.client as mqtt
   import json
   import csv
   from datetime import datetime
   import os

   CSV_FILE = "sensor_data.csv"
   FIELDS = ["timestamp", "site_id", "temperature", "humidity", "light", "alarm"]

   def init_csv():
       if not os.path.exists(CSV_FILE):
           with open(CSV_FILE, "w", newline="") as f:
               writer = csv.DictWriter(f, fieldnames=FIELDS)
               writer.writeheader()

   def on_message(client, userdata, msg):
       payload = json.loads(msg.payload.decode())
       timestamp = datetime.now().isoformat()
       row = {
           "timestamp": timestamp,
           "site_id": msg.topic.split("/")[1],
           "temperature": payload.get("temperature", ""),
           "humidity": payload.get("humidity", ""),
           "light": payload.get("light", ""),
           "alarm": payload.get("alarm", "NORMAL"),
       }
       with open(CSV_FILE, "a", newline="") as f:
           writer = csv.DictWriter(f, fieldnames=FIELDS)
           writer.writerow(row)

   init_csv()
   client = mqtt.Client()
   client.on_message = on_message
   client.connect("localhost", 1883, 60)
   client.subscribe("site/+/telemetry")
   client.loop_forever()
   ```

2. รัน Data Logger อย่างน้อย 15-20 นาที โดยสร้างเหตุการณ์จำลอง (เป่า DHT11, บัง LDR, เปิด PIR)
3. ตรวจสอบไฟล์ `sensor_data.csv`

   ```bash
   head -20 sensor_data.csv
   ```

4. นำเข้าข้อมูลใน Python เพื่อตรวจสอบ

   ```python

   df = pd.read_csv("sensor_data.csv")
   print(df.describe())
   print(df["alarm"].value_counts())
   ```

# การทดลองที่ 6.2 การวิเคราะห์แนวโน้ม Fault ด้วย LLM

## ขั้นตอนการทดลอง

1. ติดตั้ง OpenAI Library (หรือ LLM API ที่ต้องการ)

   ```bash
   pip install openai pandas
   ```

2. เขียนโปรแกรมส่งข้อมูล Sensor ล่าสุดไปยัง LLM เพื่อวิเคราะห์

   ```python
   import pandas as pd
   import json
   from openai import OpenAI

   client = OpenAI(
       api_key="YOUR_API_KEY",
       base_url="https://api.openai.com/v1"  # หรือ endpoint ของ LLM ที่ใช้
   )

   def analyze_site_data(site_id, data):
       prompt = f"""
       You are a telecom base station monitoring AI.
       Analyze the following sensor data from Site {site_id}:

       Recent readings:
       {json.dumps(data, indent=2)}

       Provide analysis in this format:
       1. Current Status: (Normal/Warning/Critical)
       2. Detected Anomalies:
       3. Possible Causes:
       4. Recommended Action:
       """

       response = client.chat.completions.create(
           model="gpt-4o-mini",  # หรือ model ที่ต้องการ
           messages=[{"role": "user", "content": prompt}],
           temperature=0.3,
       )
       return response.choices[0].message.content

   # อ่านข้อมูลล่าสุดจาก CSV
   df = pd.read_csv("sensor_data.csv")
   latest = df.tail(5).to_dict("records")

   result = analyze_site_data("A-01", latest)
   print(result)
   ```

3. ทดสอบส่งข้อมูลใน 3 สถานการณ์
   - ข้อมูลปกติ (Normal)
   - อุณหภูมิสูงขึ้นผิดปกติ (Cooling Failure)
   - ทั้งอุณหภูมิสูง + ความชื้นสูง (Power Problem)

# การทดลองที่ 6.3 การสร้าง Fault Prediction Alert

## ขั้นตอนการทดลอง

1. สร้างระบบแจ้งเตือนอัตโนมัติเมื่อ LLM ตรวจพบความผิดปกติ

   ```python
   def check_and_alert():
       df = pd.read_csv("sensor_data.csv")
       if len(df) < 3:
           return

       latest = df.tail(3).to_dict("records")
       analysis = analyze_site_data("A-01", latest)

       if "Critical" in analysis or "Warning" in analysis:
           # ส่ง Alert ผ่าน MQTT
           alert_payload = {
               "site_id": "A-01",
               "timestamp": datetime.now().isoformat(),
               "analysis": analysis,
               "action_required": "Critical" in analysis,
           }
           mqtt_client.publish("site/a01/alert", json.dumps(alert_payload))
           print("ALERT SENT:", analysis[:100])
       else:
           print("Status: Normal")
   ```

2. ตั้งเวลาเรียก `check_and_alert()` ทุก 30 วินาที
3. ทดสอบโดยสร้างสถานการณ์ผิดปกติและสังเกต Alert ที่ถูกส่งผ่าน MQTT
