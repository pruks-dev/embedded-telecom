# การทดลองที่ 2.1 การเชื่อมต่อ WiFi และส่งข้อมูลแบบ TCP/IP

## ขั้นตอนการทดลอง

1. เขียนโปรแกรมเชื่อมต่อ ESP32 เข้ากับ WiFi

   ```cpp
   #include <WiFi.h>

   const char* ssid = "YOUR_SSID";
   const char* password = "YOUR_PASSWORD";

   WiFiClient client;

   void setup() {
     Serial.begin(115200);
     WiFi.begin(ssid, password);

     while (WiFi.status() != WL_CONNECTED) {
       delay(500);
       Serial.print(".");
     }
     Serial.println("\nConnected: " + WiFi.localIP().toString());
   }

   void loop() {
     // TCP Client test
     if (client.connect("monitoring-center.local", 5000)) {
       client.println("HELLO FROM SITE A-01");
       client.stop();
     }
     delay(10000);
   }
   ```

2. บันทึก IP Address ที่ ESP32 ได้รับ (DHCP)
3. ทดสอบ Ping จากคอมพิวเตอร์ไปยัง ESP32

   ```bash
   ping <ESP32_IP>
   ```

4. สังเกตการเชื่อมต่อ — ตรวจสอบ Latency และ Packet Loss

# การทดลองที่ 2.2 การสร้าง HTTP REST API

## ขั้นตอนการทดลอง

1. เขียนโปรแกรม ESP32 เป็น HTTP Server ส่งข้อมูล Sensor ในรูปแบบ JSON

   ```cpp
   #include <WebServer.h>
   #include <ArduinoJson.h>

   WebServer server(80);

   void handleTelemetry() {
     StaticJsonDocument<200> doc;
     doc["site_id"] = "A-01";
     doc["temperature"] = dht.readTemperature();
     doc["humidity"] = dht.readHumidity();
     doc["light"] = analogRead(LDRPIN);

     String response;
     serializeJson(doc, response);
     server.send(200, "application/json", response);
   }

   void setup() {
     // ... WiFi connection same as 2.1 ...
     server.on("/api/telemetry", handleTelemetry);
     server.begin();
   }

   void loop() {
     server.handleClient();
   }
   ```

2. ทดสอบด้วย curl หรือเว็บเบราว์เซอร์

   ```bash
   curl http://<ESP32_IP>/api/telemetry
   ```

3. ควรได้ผลลัพธ์

   ```json
   {"site_id":"A-01","temperature":30.5,"humidity":65.0,"light":2800}
   ```

# การทดลองที่ 2.3 การส่งข้อมูลด้วย MQTT Telemetry

## ขั้นตอนการทดลอง

1. ติดตั้ง MQTT Broker (Mosquitto) บนคอมพิวเตอร์

   ```bash
   # macOS
   brew install mosquitto
   mosquitto -d

   # Linux
   sudo apt install mosquitto mosquitto-clients
   sudo systemctl start mosquitto
   ```

2. ติดตั้ง MQTT Library สำหรับ ESP32 (`PubSubClient`)
3. เขียนโปรแกรม ESP32 ส่งข้อมูล Sensor ผ่าน MQTT

   ```cpp
   #include <PubSubClient.h>

   WiFiClient espClient;
   PubSubClient mqtt(espClient);
   const char* mqtt_server = "<PC_IP_ADDRESS>";

   void setup() {
     // ... WiFi connection ...
     mqtt.setServer(mqtt_server, 1883);
   }

   void reconnect() {
     while (!mqtt.connected()) {
       if (mqtt.connect("SiteA01")) {
         mqtt.subscribe("site/a01/command");
       }
     }
   }

   void loop() {
     if (!mqtt.connected()) reconnect();
     mqtt.loop();

     StaticJsonDocument<200> doc;
     doc["temperature"] = dht.readTemperature();
     doc["humidity"] = dht.readHumidity();
     doc["light"] = analogRead(LDRPIN);

     String payload;
     serializeJson(doc, payload);
     mqtt.publish("site/a01/telemetry", payload.c_str());

     delay(5000);
   }
   ```

4. ทดสอบ Subscribe ข้อมูลจากคอมพิวเตอร์

   ```bash
   mosquitto_sub -h localhost -t "site/a01/telemetry"
   ```

5. ควรเห็น JSON payload ทุก 5 วินาที

## บันทึกผลการทดลอง

| ครั้งที่ | MQTT Topic | Payload ที่ได้รับ | QoS |
| ------- | ---------- | ---------------- | --- |
| 1 | site/a01/telemetry | | 0 |
| 2 | site/a01/telemetry | | 0 |
| 3 | site/a01/telemetry | | 0 |

# การทดลองที่ 2.4 การสร้าง Heartbeat

## ขั้นตอนการทดลอง

1. เพิ่ม Heartbeat Mechanism ในโปรแกรม ESP32 เพื่อแจ้งสถานะว่า Site ยังทำงานปกติ

   ```cpp
   void sendHeartbeat() {
     StaticJsonDocument<100> doc;
     doc["site_id"] = "A-01";
     doc["status"] = "online";
     doc["uptime"] = millis() / 1000;
     doc["rssi"] = WiFi.RSSI();

     String payload;
     serializeJson(doc, payload);
     mqtt.publish("site/a01/heartbeat", payload.c_str());
   }
   ```

2. ส่ง Heartbeat ทุก 30 วินาที
3. ตรวจสอบว่าหาก ESP32 หลุดจากการเชื่อมต่อ WiFi จะไม่มีการส่ง Heartbeat

## บันทึกผลการทดลอง

| ช่วงเวลา | Heartbeat Payload | WiFi RSSI (dBm) | Status |
| --------- | ----------------- | --------------- | ------ |
| T = 30s | | | |
| T = 60s | | | |
| T = 90s | | | |

### คำถามท้ายการทดลอง

1. MQTT แตกต่างจาก HTTP อย่างไรในการส่งข้อมูล Telemetry
2. ถ้า MQTT Broker ล่ม จะมีผลต่อการทำงานของ Sensor Node หรือไม่
3. การใช้ Heartbeat มีประโยชน์อย่างไรในการตรวจสอบสถานะ Site

# คำถามท้ายใบงาน

1. จงเปรียบเทียบข้อดีข้อเสียระหว่าง HTTP REST API และ MQTT สำหรับงาน Telecommunication Monitoring
2. ในระบบจริงที่มีหลาย Site (หลายร้อย Site) ควรออกแบบ MQTT Topic อย่างไร
3. Heartbeat Interval ควรตั้งค่าเท่าใดจึงจะเหมาะสม — พิจารณาทั้ง Network Traffic และความแม่นยำ
4. ถ้าต้องการความปลอดภัยของข้อมูล ควรเพิ่มกลไกอะไรในการส่งข้อมูล
